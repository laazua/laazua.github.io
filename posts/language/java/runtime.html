# JDK8 以后使用下面命令定制jre
someone@debian:~$ $JAVA_HOME/bin/jlink \
  --module-path $JAVA_HOME/jmods \
  --add-modules java.base,java.logging,java.sql \
  --output jre
## 其中 --add-modules 可以添加需要的模块
